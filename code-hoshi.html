JavaScript

// Gestion du panier
let cart = [];

// Ajouter un produit au panier
function addToCart(product) {
  cart.push(product);
  updateCartBadge();
  saveCart();
}

// Mettre √† jour le badge du panier
function updateCartBadge() {
  const badge = document.getElementById('cartBadge');
  badge.textContent = cart.length;
  
  if (cart.length > 0) {
    badge.classList.add('show');
  } else {
    badge.classList.remove('show');
  }
}

// Sauvegarder le panier dans localStorage
function saveCart() {
  localStorage.setItem('cart', JSON.stringify(cart));
}

// Charger le panier depuis localStorage
function loadCart() {
  const savedCart = localStorage.getItem('cart');
  if (savedCart) {
    cart = JSON.parse(savedCart);
    updateCartBadge();
  }
}

// Animation des messages de la top-bar
function rotateMessages() {
  const messages = document.querySelectorAll('.topbar-msg');
  let current = 0;
  
  setInterval(() => {
    messages[current].classList.remove('topbar-msg--visible');
    current = (current + 1) % messages.length;
    messages[current].classList.add('topbar-msg--visible');
  }, 4000);
}

// Filtrage des produits
function filterProducts(filter) {
  const products = document.querySelectorAll('.product');
  
  products.forEach(product => {
    if (filter === 'all') {
      product.style.display = 'block';
    } else {
      const stock = product.getAttribute('data-stock');
      product.style.display = stock === filter ? 'block' : 'none';
    }
  });
}

// Recherche de produits
function searchProducts(query) {
  const products = document.querySelectorAll('.product');
  const searchTerm = query.toLowerCase();
  
  products.forEach(product => {
    const title = product.querySelector('.product-title').textContent.toLowerCase();
    product.style.display = title.includes(searchTerm) ? 'block' : 'none';
  });
}

// Initialisation au chargement
document.addEventListener('DOMContentLoaded', () => {
  loadCart();
  rotateMessages();
});
